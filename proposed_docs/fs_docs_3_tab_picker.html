File Picker Quick Start

Get started quickly with our File Picker. Here are the basic implementations for popular frameworks.

1. Install the Filestack React library

npm install filestack-react

2. Use the Picker component in your app

import React from 'react';
import { PickerOverlay } from 'filestack-react';

const FileUploader = () => {
  return (
    <div>
      <h2>Filestack Uploader in React</h2>
      <PickerOverlay
        apikey={'YOUR_API_KEY'}
        onUploadDone={(result) => {
          console.log('Upload successful:', result);
        }}
        pickerOptions={{
          maxFiles: 5,
          accept: ['image/*'],
          fromSources: ['local_file_system', 'url', 'googledrive'],
        }}
      />
    </div>
  );
};

export default FileUploader;

3. Run your development server

npm start

1. Install the Filestack core JS library

npm install filestack-js

2. Import and use in your component

import { Component } from '@angular/core';
import * as filestack from 'filestack-js';

@Component({
  selector: 'app-uploader',
  template: `
    <h2>Filestack Uploader in Angular</h2>
    <button (click)="showPicker()">Upload File</button>
  `,
})
export class UploaderComponent {
  // Initialize the client with your API key
  private readonly client: filestack.Client = filestack.init('YOUR_API_KEY');

  showPicker() {
    const options: filestack.PickerOptions = {
      onUploadDone: (res) => console.log(res),
      maxFiles: 5,
    };
    this.client.picker(options).open();
  }
}

3. Run your development server

ng serve

1. Create an HTML file (e.g., index.html) with the following code

<!DOCTYPE html>
<html>
<head>
  <title>Filestack Demo</title>
</head>
<body>
  <!-- 1. Add the Filestack script to your HTML -->
  <script src="//static.filestackapi.com/filestack-js/3.x.x/filestack.min.js"></script>

  <!-- 2. Add a button to trigger the picker -->
  <button id="picker">Upload File</button>

  <!-- 3. Initialize the picker with your API key -->
  <script>
    const client = filestack.init('YOUR_API_KEY');
    const options = {
      onUploadDone: (res) => console.log(res),
    };
    const picker = client.picker(options);

    const pickerBtn = document.getElementById('picker');
    pickerBtn.addEventListener('click', () => picker.open());
  </script>
</body>
</html>

2. Run the project

Simply open the `index.html` file directly in your web browser to see it in action.
