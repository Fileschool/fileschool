Quick and Complete File Uploads in Your Angular App

Use the official Filestack component library to add a powerful file picker to your Angular application with minimal setup.

Angular Quick Setup

Project Setup

Install the Filestack Angular SDK into your existing Angular project.

npm install filestack-angular

Import the Module

Next, import the `FilestackModule` into your `app.module.ts`.


import { FilestackModule } from 'filestack-angular';
// ... other imports

@NgModule({
  declarations: [
    // ... your components
  ],
  imports: [
    BrowserModule,
    FilestackModule.forRoot({ apikey: 'YOUR_API_KEY' }), // Add your API key here
    // ... other modules
  ],
  providers: [],
  bootstrap: [AppComponent]
})
export class AppModule { }

Implement the Picker

Use the `filestack` component in your template and handle the output events in your component's class.


// In your-component.component.html

<h1>My Awesome Angular App</h1>

<filestack
  [pickerOptions]="pickerOptions"
  (uploadSuccess)="onUploadSuccess($event)"
  (uploadError)="onUploadError($event)">
</filestack>
                            

// In your-component.component.ts

import { Component } from '@angular/core';
import { PickerOptions } from 'filestack-js';

@Component({
  selector: 'app-root',
  templateUrl: './app.component.html',
})
export class AppComponent {
  
  // Optional: configure the picker
  pickerOptions: PickerOptions = {
    accept: 'image/*',
    fromSources: ['local_file_system', 'url', 'googledrive'],
    maxFiles: 5,
  };

  onUploadSuccess(result: any) {
    console.log('Upload successful:', result);
  }

  onUploadError(error: any) {
    console.error('Upload error:', error);
  }
}

Start Your Application

Run the standard Angular CLI command to serve your application.

ng serve

Your browser will open to `http://localhost:4200`, where the Filestack button will be rendered.
