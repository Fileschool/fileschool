Complete Froala Config

The actual configuration used in this demo, combining all the tips above:

new FroalaEditor('#comment-editor', {
  // Toolbar with emoticons
  toolbarButtons: {
    moreText: {
      buttons: ['bold', 'italic', 'emoticons', 
               'underline', 'textColor', 
               'paragraphFormat']
    },
    moreParagraph: {
      buttons: ['alignLeft', 'alignCenter', 
               'alignRight', 'formatOL', 'formatUL']
    },
    moreRich: {
      buttons: ['insertLink', 'insertImage', 
               'insertTable']
    },
    moreMisc: {
      buttons: ['undo', 'redo']
    }
  },
  
  // Character counter settings
  charCounterCount: true,
  charCounterMax: 2000,
  
  // Quick insert configuration
  quickInsertEnabled: true,
  quickInsertButtons: ['image', 'video', 
                      'embedly', 'table', 
                      'ul', 'ol', 'hr'],
  quickInsertTags: ['p', 'div', 'h1', 'h2', 
                   'h3', 'h4', 'h5', 'h6'],
  
  // General settings
  placeholderText: 'Share your thoughts...',
  heightMin: 200,
  heightMax: 400,
  
  events: {
    'initialized': function() {
      console.log('Blog editor initialized');
    },
    'contentChanged': function() {
      console.log('Content changed');
    }
  }
});
